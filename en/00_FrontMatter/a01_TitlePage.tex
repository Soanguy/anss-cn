  % Table of contents

  \starttitle
    [title={Table of Contents}]

    % Header
    \setupheadertexts
      [\tfx Table of Contents]
      [\tfx\pagenumber]

    % Format of entries in the ToC
    \setuplist
      [part]
      [
        before={\blank[2*big]},
        after={\blank[big]},
        style=\bfa,
        aligntitle=yes,
      ]

    \setuplist
      [chapter]
      [
        before=\blank,
        style=\bf,
        margin=.5cm,
        aligntitle=yes
      ]

    \setuplist
      [section]
      [
        aligntitle=yes,
        style=\tfx,
        margin=1.5cm
      ]

    % Let us define our list (called a Toc)
    \definecombinedlist
      [Toc]
      [part, chapter, section]
      [level=subsection, alternative=c]

    % and insert it in the context of smaller interline space
    \start
      \setupwhitespace[none]
      \switchtobodyfont[11pt]
      \setupinterlinespace[small]
      \placeToc
    \stop

  \stoptitle
