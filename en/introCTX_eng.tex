    \startfrontmatter
        \component 00_FrontMatter/a01_TitlePage

            \setupheadnumber[chapter][0]

            \component 01_BodyMatter/b03_Commands
            \component 01_BodyMatter/b04_SourceFile
        \stoppart

        % Second Part: Global aspects
        \startpart
            [title=Global aspects of the document]

            % I am readjusting the chapter counter, since the part command has

            % chapter numbers should follow in sequence throughout the entire
            % document.

            \component 01_BodyMatter/b05_Pages
            \component 01_BodyMatter/b06_Fonts
            \component 01_BodyMatter/b07_Structure
            \component 01_BodyMatter/b08_Lists

            \component 01_BodyMatter/b10_CharText
            \component 01_BodyMatter/b11_ParasLines
            \component 01_BodyMatter/b12_SpecialParas
            \component 01_BodyMatter/b13_Floats
        \stoppart

        % Last Part: Appendices
        \setuphead[part][number=no] % No numbering for this part

        \startpart
            [title=Appendices]

            % Readjust the headers
            \setupheadertexts [{\tfx Appendix \getmarking[chapternumber]}]
                              [{\tfx\hbox to 2em{\hss\pagenumber}}]

            \setuplabeltext[en][chapter=Appendix ]
            \setuphead[chapter][conversion=A]

            \component 02_Appendices/Installation
            \component 02_Appendices/Symbols
            \component 02_Appendices/Macros
